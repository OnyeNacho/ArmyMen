
// Fruin's ai-controlled unit support for multiplayer desert2 missions

IF startup THEN
	TRIGGER setupgreen,
	TRIGGER setuptan,
	TRIGGER setupblue,
	TRIGGER	setupgray

IF setupgreen THEN
	SETAICONTROL GreenGrenadier1,
	SETAICONTROL GreenGrenadier2,
	SETAICONTROL GreenBazooka1,
	SETAICONTROL GreenBazooka2,
	SETAICONTROL GreenBazooka3, 
	SETAICONTROL GreenMortar1,
	SETAICONTROL GreenMortar2,
	SETAICONTROL GreenRifle1,
	ORDER GreenGrenadier1 GOTO GreenGrenadier1 INMODE KNEELFIRE THEN ATMODE NESTFIRE,
	ORDER GreenGrenadier2 GOTO GreenGrenadier2 INMODE KNEELFIRE THEN ATMODE NESTFIRE,
	ORDER GreenBazooka1 GOTO GreenBazooka1 INMODE KNEELFIRE THEN ATMODE NESTFIRE,
	ORDER GreenBazooka2 GOTO GreenBazooka2 INMODE KNEELFIRE THEN ATMODE NESTFIRE,
	ORDER GreenBazooka3 GOTO GreenBazooka3 INMODE IGNORE THEN ATMODE NESTFIRE,
	ORDER GreenMortar1 GOTO GreenMortar1 INMODE KNEELFIRE THEN ATMODE KNEELFIRE,
	ORDER GreenMortar2 GOTO GreenMortar2 INMODE KNEELFIRE THEN ATMODE KNEELFIRE,
	ORDER GreenRifle1 GOTO GreenRifle1 INMODE IGNORE THEN ATMODE NESTFIRE
