// enable mash tents
IF startup THEN TRIGGERDELAY 0.1 setmash
IF setmash THEN 
TRIGGER greenmashready,
TRIGGER tanmashready,
TRIGGER bluemashready,
TRIGGER graymashready

IF on_gsmedpad AFTER off_gsmedpad AND gsmedtime AFTER NULL 
AND greenmashready AFTER killed_greenmash THEN

TRIGGERDELAY 0.5 gsmedtime

IF on_tsmedpad AFTER off_tsmedpad AND tsmedtime AFTER NULL 
AND tanmashready AFTER killed_tanmash THEN

AND graymashready AFTER killed_graymash THEN
