



PAD Helipad		99	GreenSarge	> 0



//ENDGAMES
IF killed_GreenSarge THEN
	ENDSOUNDLOOP CHOPPER_LOOP,
	TRIGGERDELAY 1 tanwins

IF Gone THEN
	ENDSOUNDLOOP CHOPPER_LOOP,
	TRIGGER greenwins



//INIT
IF startup THEN
	ALLY BLUE GREEN,
	ALLY GREEN BLUE,
	VULNERABLESARGE,
	ORDER TanMortar3 GOTO TanMortar3 INMODE IGNORE THEN ATMODE NESTFIRE,
	ORDER TanMortar4 GOTO TanMortar4 INMODE IGNORE THEN ATMODE NESTFIRE,
	ORDER TanRifle33 GOTO TanRifle33 INMODE IGNORE THEN ATMODE STANDFIRE,
	ORDER TanRifle34 GOTO TanRifle34 INMODE IGNORE THEN ATMODE STANDFIRE,
	ORDER TanRifle35 GOTO TanRifle35 INMODE IGNORE THEN ATMODE ATTACK,
	ORDER TanGrenadier7 GOTO TanGrenadier7 INMODE IGNORE THEN ATMODE ATTACK,
	ORDER TanGrenadier9 GOTO TanGrenadier9 INMODE IGNORE THEN ATMODE ATTACK,
	SETAICONTROL BlueJeep1,
	ORDER BlueJeep1 GOTO (3970, 2425) INMODE IGNORE THEN TRIGGER JeepOff,
	SETAICONTROL GreenHalftrack3,
	ORDER GreenHalftrack3 GOTO (3970, 2550) INMODE IGNORE THEN TRIGGER TruckOff

IF JeepOff THEN
	UNDEPLOY BlueJeep1
IF TruckOff THEN
	UNDEPLOY GreenHalftrack3


//GAME EVENTS

IF FlySound THEN
	PLAYSOUNDLOOP CHOPPER_LOOP

IF LandSound THEN
	ENDSOUNDLOOP CHOPPER_LOOP,
	PLAYSTREAM CHOPPER_SPIN_DOWN

IF LeaveSound THEN
	PLAYSTREAM CHOPPER_SPIN_UP

IF INORDER AllDead on_Helipad THEN
	TRIGGER GoChopper

IF PlaceLure THEN
	REMOVEOBJECT CopterLure,
	PLACEOBJECT CopterLure Chopper

IF get_GreenSarge_CopterLure THEN
	INVULNERABLESARGE,
	UNDEPLOY GreenSarge,
	TRIGGER Leave

IF ALLOF killed_TanMortar3 killed_TanMortar4 killed_TanRifle33 killed_TanRifle34 killed_TanGrenadier7 killed_TanGrenadier9 killed_TanRifle35 THEN
	TRIGGER AllDead
